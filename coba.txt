<script setup>
// --- LOGIKA JAVASCRIPT ---
// (INI SEMUA SAMA PERSIS SEPERTI SEBELUMNYA, TIDAK ADA PERUBAHAN)

import { ref, onMounted, watch, nextTick } from "vue"

const messages = ref([])
const userInput = ref("")
const isLoading = ref(false)
const chatContainer = ref(null)

onMounted(() => {
  messages.value.push({
    role: "bot",
    type: "text",
    content: "Halo! Saya adalah AI asisten bank. Apa yang ingin Anda ketahui?",
  })
})

async function handleSubmit() {
  if (!userInput.value.trim()) return
  isLoading.value = true
  const currentMessage = userInput.value

  messages.value.push({ role: "user", type: "text", content: currentMessage })
  userInput.value = ""

  const requestPayload = { prompt: currentMessage }

  try {
    const response = await fetch("http://localhost:8080/api/query", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify(requestPayload),
    })
    const data = await response.json()
    if (!response.ok) {
      throw new Error(data.error || "Terjadi kesalahan dari API")
    }
    messages.value.push({ role: "bot", type: "data", data: data })
  } catch (error) {
    let errorMsg = error.message
    if (error.message.includes("Failed to fetch")) {
      errorMsg =
        "Gagal terhubung ke backend (http://localhost:8080). Pastikan backend Go kamu sudah jalan!"
    }
    messages.value.push({ role: "bot", type: "error", content: errorMsg })
  }
  isLoading.value = false
}

const formatter = new Intl.NumberFormat("id-ID", {
  style: "currency",
  currency: "IDR",
  minimumFractionDigits: 0,
  maximumFractionDigits: 0,
})

watch(
  messages,
  () => {
    nextTick(() => {
      if (chatContainer.value) {
        chatContainer.value.scrollTop = chatContainer.value.scrollHeight
      }
    })
  },
  { deep: true }
)
</script>

                  />
                </svg>
                <span>{{ message.content }}</span>
              </div>
            </template>
            <template v-else-if="message.type === 'data'">
              <div
                v-if="!message.data || message.data.length === 0"
                class="no-data"
              >
                <svg
                  class="empty-icon"

                  stroke-width="1.5"

                  <path

                    d="M12 9.75 14.25 12 12 14.25l-2.25-2.25L12 9.75ZM12 18a2.25 2.25 0 1 0 0-4.5A2.25 2.25 0 0 0 12 18ZM12 2.25V18"
                  />
                </svg>
                <p>Tidak ada data ditemukan.</p>
              </div>
              <div v-else class="table-wrapper">
                <table class="data-table">
                  <thead>
                    <tr>
                      <th
                        v-for="headerKey in Object.keys(message.data[0])"
                        :key="headerKey"
                      >
                        {{
                          headerKey
                            .replace(/_/g, " ")
                            .replace(/\b\w/g, c => c.toUpperCase())
                        }}
                      </th>
                    </tr>
                  </thead>
                  <tbody>
                    <tr
                      v-for="(rowData, rIndex) in message.data"
                      :key="rIndex"
                      class="table-row"
                    >
                      <td
                        v-for="headerKey in Object.keys(rowData)"
                        :key="headerKey"
                      >
                        <template
                          v-if="
                            headerKey.includes('saldo') ||
                            headerKey.includes('jumlah') ||
                            headerKey.includes('debit') ||
                            headerKey.includes('kredit')
                          "
                        >
                          <span class="number-cell">{{
                            formatter.format(
                              parseFloat(rowData[headerKey]) || 0
                            )
                          }}</span>
                        </template>
                        <template
                          v-else-if="
                            headerKey.includes('waktu_transaksi') ||
                            headerKey.includes('tanggal')
                          "
                        >
                          {{
                            new Date(rowData[headerKey]).toLocaleString(
                              "id-ID",
                              { dateStyle: "short", timeStyle: "short" }
                            )
                          }}
                        </template>
                        <template v-else>
                          {{ rowData[headerKey] }}
                        </template>
                      </td>
                    </tr>
                  </tbody>
                </table>
              </div>
            </template>

        </div>
      </div>

        <input

          type="text"

          v-model="userInput"
          :disabled="isLoading"
          autocomplete="off"
